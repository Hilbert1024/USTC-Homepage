func4(int a, int b, int c) {
    int result = c
    result -= b
    int tmp = result
    tmp >>= 31 
    result += tmp
    result >>= 1
    tmp = b + result // 至此tmp = 7，令 a == tmp 即可
    if (tmp > a) {
        c = tmp - 1
        result = func4(a, b, c)
        result += result
        return result
    }
    result = 0
    if (tmp == a) {
        return result // 跳转至此
    }
    b = tmp + 1
    result = func4(a, b, c)
    result = 2 * result + 1
    return result
